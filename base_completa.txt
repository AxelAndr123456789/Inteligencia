-- Crear base de datos
CREATE DATABASE qbd_farmacia_magistral;

-- ============================================================
-- ELIMINAR TABLAS SI EXISTEN
-- ============================================================
IF OBJECT_ID('FACT_VentaFM', 'U') IS NOT NULL DROP TABLE FACT_VentaFM;

IF OBJECT_ID('DIM_FormulaMagistral', 'U') IS NOT NULL DROP TABLE DIM_FormulaMagistral;
IF OBJECT_ID('DIM_ProductoTerminado', 'U') IS NOT NULL DROP TABLE DIM_ProductoTerminado;
IF OBJECT_ID('DIM_Cliente', 'U') IS NOT NULL DROP TABLE DIM_Cliente;
IF OBJECT_ID('DIM_Sede', 'U') IS NOT NULL DROP TABLE DIM_Sede;

IF OBJECT_ID('DIM_Tiempo_FM', 'U') IS NOT NULL DROP TABLE DIM_Tiempo_FM;

GO

-- ============================================================
-- TABLAS DIMENSIONALES
-- ============================================================

-- CREDENCIALES DE USUARIO (Login)
CREATE TABLE DIM_Usuario (
    UsuarioKey     INT PRIMARY KEY,
    Username       VARCHAR(50) NOT NULL UNIQUE,
    PasswordHash   VARCHAR(255) NOT NULL,
    Nombres        VARCHAR(100) NOT NULL,
    Apellidos      VARCHAR(100) NOT NULL,
    Email          VARCHAR(100),
    Rol            VARCHAR(20) NOT NULL,
    Activo         BIT NOT NULL DEFAULT 1
);
GO

-- TIEMPO PARA FÓRMULAS MAGISTRALES
CREATE TABLE DIM_Tiempo_FM (
    FechaKey       INT PRIMARY KEY,
    Fecha          DATE NOT NULL,
    Anio           INT NOT NULL,
    Trimestre      INT NOT NULL,
    Mes            INT NOT NULL,
    MesNombre      VARCHAR(20) NOT NULL,
    Semana         INT NOT NULL,
    DiaSemana      VARCHAR(15) NOT NULL,
    EsFinDeSemana  BIT NOT NULL,
    Periodo        VARCHAR(20) NOT NULL
);
GO

-- TIEMPO PARA PRODUCTOS TERMINADOS
CREATE TABLE DIM_Tiempo_PT (
    FechaKey       INT PRIMARY KEY,
    Fecha          DATE NOT NULL,
    Anio           INT NOT NULL,
    Trimestre      INT NOT NULL,
    Mes            INT NOT NULL,
    MesNombre      VARCHAR(20) NOT NULL,
    Semana         INT NOT NULL,
    DiaSemana      VARCHAR(15) NOT NULL,
    EsFinDeSemana  BIT NOT NULL,
    Periodo        VARCHAR(20) NOT NULL
);
GO

-- CLIENTES
CREATE TABLE DIM_Cliente (
    ClienteKey      INT PRIMARY KEY,
    DniCliente      VARCHAR(8) NOT NULL,
    NombresC        VARCHAR(100) NOT NULL,
    ApellidosC      VARCHAR(100) NOT NULL,
    FechaNacimiento DATE,
    CelularC        VARCHAR(15),
    ApoderadoC      VARCHAR(150)
);
GO

-- SEDES
CREATE TABLE DIM_Sede (
    SedeKey      INT PRIMARY KEY,
    NombreSede   VARCHAR(100) NOT NULL,
    Ciudad       VARCHAR(50) NOT NULL,
    Direccion    VARCHAR(200) NOT NULL,
    Region       VARCHAR(50) NOT NULL
);
GO

-- MÉDICOS
CREATE TABLE DIM_Medico (
    MedicoKey          INT PRIMARY KEY,
    ColegiaturaMedico  VARCHAR(20) NOT NULL,
    NombresM           VARCHAR(100) NOT NULL,
    ApellidosM         VARCHAR(100) NOT NULL,
    CelularM           VARCHAR(15),
    EstadoM            VARCHAR(20) NOT NULL
);
GO

-- FÓRMULAS MAGISTRALES
CREATE TABLE DIM_FormulaMagistral (
    CodFormula       VARCHAR(20) PRIMARY KEY,
    DescripcionFM    VARCHAR(300) NOT NULL,
    UnidadMedidaFM   VARCHAR(30) NOT NULL,
    CostoUnitarioFM  DECIMAL(10,2) NOT NULL,
    EstadoFM         VARCHAR(20) NOT NULL
);
GO

-- PRODUCTOS TERMINADOS
CREATE TABLE DIM_ProductoTerminado (
    CodProducto       VARCHAR(20) PRIMARY KEY,
    DescripcionPT     VARCHAR(300) NOT NULL,
    DescripcionUso    VARCHAR(500) NOT NULL,
    UnidadMedidaPT    VARCHAR(30) NOT NULL,
    CostoUnitarioPT   DECIMAL(10,2) NOT NULL,
    StockPT           INT NOT NULL,
    EstadoPT          VARCHAR(20) NOT NULL
);
GO

-- ============================================================
-- TABLAS DE HECHOS
-- ============================================================

-- VENTAS FÓRMULAS MAGISTRALES
CREATE TABLE FACT_VentaFM (
    CodVenta         VARCHAR(20) PRIMARY KEY,
    CodFormula       VARCHAR(20) NOT NULL,
    FechaKey         INT NOT NULL,
    ClienteKey       INT NOT NULL,
    SedeKey          INT NOT NULL,
    MedicoKey        INT NOT NULL,
    Medidas          VARCHAR(100),
    Cantidad         INT NOT NULL,
    CostoUnitarioFM  DECIMAL(10,2) NOT NULL,
    SubTotalFM       DECIMAL(10,2) NOT NULL,
    CONSTRAINT FK_VentaFM_Formula  FOREIGN KEY (CodFormula)  REFERENCES DIM_FormulaMagistral(CodFormula),
    CONSTRAINT FK_VentaFM_Tiempo   FOREIGN KEY (FechaKey)    REFERENCES DIM_Tiempo_FM(FechaKey),
    CONSTRAINT FK_VentaFM_Cliente  FOREIGN KEY (ClienteKey)  REFERENCES DIM_Cliente(ClienteKey),
    CONSTRAINT FK_VentaFM_Sede     FOREIGN KEY (SedeKey)     REFERENCES DIM_Sede(SedeKey),
    CONSTRAINT FK_VentaFM_Medico   FOREIGN KEY (MedicoKey)   REFERENCES DIM_Medico(MedicoKey)
);
GO

-- VENTAS PRODUCTOS TERMINADOS
CREATE TABLE FACT_VentaPT (
    CodVenta          VARCHAR(20) PRIMARY KEY,
    CodProducto       VARCHAR(20) NOT NULL,
    FechaKey          INT NOT NULL,
    ClienteKey        INT NOT NULL,
    SedeKey           INT NOT NULL,
    Medidas           VARCHAR(100),
    Cantidad          INT NOT NULL,
    CostoUnitarioPT   DECIMAL(10,2) NOT NULL,
    SubTotalPT        DECIMAL(10,2) NOT NULL,
    StockDespues      INT NOT NULL,
    CONSTRAINT FK_VentaPT_Producto FOREIGN KEY (CodProducto) REFERENCES DIM_ProductoTerminado(CodProducto),
    CONSTRAINT FK_VentaPT_Tiempo   FOREIGN KEY (FechaKey)    REFERENCES DIM_Tiempo_PT(FechaKey),
    CONSTRAINT FK_VentaPT_Cliente  FOREIGN KEY (ClienteKey)  REFERENCES DIM_Cliente(ClienteKey),
    CONSTRAINT FK_VentaPT_Sede     FOREIGN KEY (SedeKey)     REFERENCES DIM_Sede(SedeKey)
);
GO
